154void *
155receive_from_client(void *arg) {
156 thread_args *t = (thread_args *)arg;
157#ifdef DEBUG_FLAG
158 prtlog("执行 receive_from_client(). ");
159 prtlog("client_fd = %d", client_fd);
160#endif
161 memset(received_message, 0, kStringLength);
162 bool is_name = 1;
163 do {
164 int ret = recv(t->client_fd, received_message, sizeof(received_message), 0);
165#ifdef DEBUG_FLAG
166 prtlog("recv()已经被执行. ");
167#endif
168 if (ret < 0) {
169 prterr(recv());
170 close(t->client_fd);
171 free(t);
172 return NULL;
173 } else if (ret == 0) {
174 disconnect_client(t->client_fd);
175 free(t);
176 return NULL;
177 }
178 prtlog("\033[32m成功接收一条来自客户端 %d 的消息: \033[0m%s", t->client_fd, received_message);
179 // 来自客户端的第一条消息被认为是昵称.
180 if (is_name) {
181 strcpy(client_list[get_client_by_fd(t->client_fd)].name, received_message);
182 is_name = 0;
183 prtlog("该客户端的昵称设置为 %s", received_message);
184 const char msg[kStringLength] = " 进入了聊天室. ";
185 strcat(received_message, msg);
186 forward_all_client(-1, received_message);
187 } else { // 往后才是正式的消息.
188 prtlog("尝试群发这条消息. ");
189 if (forward_all_client(get_client_by_fd(t->client_fd), received_message) < 0) {
190 prtlog("\033[31m群发消息时发生异常. \033[0m");
191 }
192 prtlog("\033[32m群发成功. \033[0m");
193 }
194 } while (t->client_fd > 0 && client_count < kMaxClientCount);
195 free(t);
196 return NULL;
197}

256 client_list[client_count].thread_id = pthread_self();
