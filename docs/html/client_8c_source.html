127 bool is_username = 1;

139 }
140 if (is_username) {
141 pthread_mutex_lock(&lock);

143 pthread_mutex_unlock(&lock);
144 is_username = 0;

146 pthread_mutex_lock(&lock);
147 prtlog("%s", received_message);

149 pthread_mutex_unlock(&lock);

151 }
152 } while (!exit_flag);
